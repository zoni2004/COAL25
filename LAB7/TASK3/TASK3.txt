.386
.MODEL flat, stdcall
.STACK 4096

ExitProcess PROTO, dwExitCode:DWORD

.data
    array1 dd 10, 20, 30, 40, 50      
    array2 dd 5, 15, 25, 35, 45       
    sum1 dd 0                         
    sum2 dd 0                        
    total dd 0                       

.code
main PROC
    push OFFSET array1
    call sum_array
    mov sum1, eax
    
    push OFFSET array2
    call sum_array
    mov sum2, eax
    
    mov eax, sum1
    add eax, sum2
    mov total, eax
    
    push 0
    call ExitProcess
main ENDP

sum_array PROC
    push ebp
    mov ebp, esp
    push ebx
    push ecx
    
    mov ebx, [ebp+8]   
    mov ecx, 5         
    xor eax, eax     
    
sum_loop:
    add eax, [ebx]
    add ebx, 4          
    loop sum_loop
    
    pop ecx
    pop ebx
    mov esp, ebp
    pop ebp
    ret 4              
sum_array ENDP

END main
